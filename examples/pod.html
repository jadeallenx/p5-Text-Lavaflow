sub formatter_meta {
 my ($text) = @_;
 foreach (split /\n\n*/, $text) {
  next if(/^\s*$/ or /^\s*#/);
  s/\s*#.*$//; # we support comments
  my ($tag, $param, $option) = split /\s\s*/;
  if (! $option) {
    $option = $param;
    $param  = "";
  }
  print qq(<$tag $param>$option</$tag>\n);
 }
}
